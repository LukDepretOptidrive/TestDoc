using System;

namespace WpfTutorialSamples.XAML

	public partial class EventsSample : Window

		public EventsSample()
		{
			InitializeComponent();

		}

		private void pnlMainGrid_MouseUp(object sender, MouseButtonEventArgs e)

		}

	}
